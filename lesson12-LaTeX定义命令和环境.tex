\documentclass{article}
\usepackage{ctex}
% \newcommand - 定义命令
% 命令只能由字母组成，不能以 \end 开头
% \newcommand<命令>[<参数个数>][<首参数默认值>]{<具体定义>}
% 如定义一个不带参数的PRC命令
% 可以理解为C语言的宏定义
\newcommand\PRC{People's Republic of \emph{China}}

\newcommand\addtwonum[2][1024]{#1 + #2}  %  这里第二个方括号设置了默认参数，则后面的 #1 表示1024，如再设置多于一个的参数，会把所有的参数进行字符串拼接当作一个参数

\newcommand*{\love}[3][喜欢]{#2 #1 #3} % 这两种写法能达到相同的效果
\newcommand\loves[3][喜欢]{#2 #1 #3}
%\newcommand\abstractname{简介}
% \renewcommand 与\newcommand用法相同，但是只能作用于已有命令
\renewcommand\abstractname{内容简介}

% 定义和重定义环境
% \newenvironment{<环境名称>}[<参数个数>][<受参数默认值>]
%				 {<环境前定义>}{<环境后定义>}
% \renewenvironment{<环境名称>}[<参数个数>][<受参数默认值>]
%				 {<环境前定义>}{<环境后定义>}

% 为book类中定义摘要环境
\newenvironment{myabstract}[1][摘要]
{\small
	\begin{center}\bfseries #1\end{center}
	\begin{quotation}}
	{\end{quotation}
}

% 环境参数只有<环境前定义>中可以使用参数，
% <环境后定义>中不能再使用环境参数。
% 如果需要，可以先把前面得到的参数保存在一个命令中，在后面使用
\newenvironment{Quotation}[1]
{\newcommand\quotesource{#1} % 相当于此处定义了一个‘变量’，然后在下文中这个‘变量’展开成新环境传递的参数
	\begin{quotation}}
	{\par\hfill---《\textit{\quotesource}》
	\end{quotation}
}


\begin{document}
	\begin{abstract}
		这是一段摘要...
	\end{abstract}
	
	\begin{myabstract}[我的摘要]
		这是一段摘要...
	\end{myabstract}
	It's \PRC,I'm a Chinese.
	
	Self-defined command addtwonum \addtwonum{3}{4}.
	
	\love{猫儿}{鱼}
	
	\loves{狗}{屎}
	
	\begin{Quotation}{易$ \cdot $乾}
		初九，潜龙勿用。
	\end{Quotation}
	
\end{document}
